94 int __declspec(dllexport) __stdcall
95 ProcessKeystroke(hConIn, ir, valid, numberpad, portdebug)
96 HANDLE hConIn;
97 INPUT_RECORD *ir;
98 boolean *valid;
99 boolean numberpad;
100 int portdebug;
101 {
102  int metaflags = 0, k = 0;
103  int keycode, vk;
104  unsigned char ch, pre_ch, mk = 0;
105  unsigned short int scan;
106  unsigned long shiftstate;
107  int altseq = 0;
108  const struct pad *kpad;
109 
110  shiftstate = 0L;
111  ch = pre_ch = ir->Event.KeyEvent.uChar.AsciiChar;
112  scan = ir->Event.KeyEvent.wVirtualScanCode;
113  vk = ir->Event.KeyEvent.wVirtualKeyCode;
114  keycode = MapVirtualKey(vk, 2);
115  shiftstate = ir->Event.KeyEvent.dwControlKeyState;
116 
117  if (shiftstate & (LEFT_ALT_PRESSED|RIGHT_ALT_PRESSED)) {
118  if (ch || inmap(keycode,vk)) altseq = 1;
119  else altseq = -1; /* invalid altseq */
120  }
121  if (ch || (iskeypad(scan)) || (altseq > 0))
122  *valid = TRUE;
123  /* if (!valid) return 0; */
124  /*
125  * shiftstate can be checked to see if various special
126  * keys were pressed at the same time as the key.
127  * Currently we are using the ALT & SHIFT & CONTROLS.
128  *
129  * RIGHT_ALT_PRESSED, LEFT_ALT_PRESSED,
130  * RIGHT_CTRL_PRESSED, LEFT_CTRL_PRESSED,
131  * SHIFT_PRESSED,NUMLOCK_ON, SCROLLLOCK_ON,
132  * CAPSLOCK_ON, ENHANCED_KEY
133  *
134  * are all valid bit masks to use on shiftstate.
135  * eg. (shiftstate & LEFT_CTRL_PRESSED) is true if the
136  * left control key was pressed with the keystroke.
137  */
138  if (iskeypad(scan)) {
139  kpad = numberpad ? numpad : keypad;
140  if (shiftstate & SHIFT_PRESSED) {
141  ch = kpad[scan - KEYPADLO].shift;
142  }
143  else if (shiftstate & (LEFT_CTRL_PRESSED | RIGHT_CTRL_PRESSED)) {
144  ch = kpad[scan - KEYPADLO].cntrl;
145  }
146  else {
147  ch = kpad[scan - KEYPADLO].normal;
148  }
149  }
150  else if (altseq > 0) { /* ALT sequence */
151  if (vk == 0xBF) ch = M('?');
152  else ch = M(tolower(keycode));
153  }
154  if (ch == '\r') ch = '\n';
155 #ifdef PORT_DEBUG
156  if (portdebug) {
157  char buf[BUFSZ];
158  Sprintf(buf,
159  "PORTDEBUG (%s): ch=%u, sc=%u, vk=%d, sh=0x%X (ESC to end)",
160  shortdllname, ch, scan, vk, shiftstate);
161  fprintf(stdout, "\n%s", buf);
162  }
163 #endif
164  return ch;
165 }
