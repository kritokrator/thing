77 void
78 tty_change_color (int color, long rgb, int reverse) {
79 long inverse, working_rgb = rgb;
80 int total_power = 0, max_channel = 0;
81 int cnt = 3;
82 
83  working_rgb >>= 4;
84  while (cnt -- > 0) {
85  total_power += working_rgb & 0xf;
86  max_channel = max (max_channel, working_rgb & 0xf);
87  working_rgb >>= 8;
88  }
89 
90  if (total_power >= POWER_LIMIT ||
91  (total_power >= SECONDARY_POWER_LIMIT &&
92  max_channel >= SECONDARY_CHANNEL_LIMIT) ||
93  max_channel >= CHANNEL_LIMIT)
94  inverse = 0x000000;
95  else
96  inverse = 0xffffff;
97 
98  if (reverse) {
99  working_rgb = rgb;
100  rgb = inverse;
101  inverse = working_rgb;
102  }
103 
104  if (color >= CLR_MAX) {
105  if (color - CLR_MAX >= 5)
106  impossible ("Changing too many colors");
107  else {
108  _mt_attrs [color - CLR_MAX] [0] = rgb;
109  _mt_attrs [color - CLR_MAX] [1] = inverse;
110  _attrs_inverse [color - CLR_MAX] = reverse;
111  }
112  } else if (color >= 0) {
113  _mt_colors [color] [0] = rgb;
114  _mt_colors [color] [1] = inverse;
115  _colors_inverse [color] = reverse;
116  } else
117  impossible ("Changing negative color");
118 }
